// ==UserScript==
// @name         CF Verdict DOMjudge Style (no time/memory, domjudge colors)
// @namespace    http://tampermonkey.net/
// @version      1.3
// @description  DOMjudge-like verdicts on Codeforces: hide test numbers, hide time/memory, custom colors.
// @author       Maxmilite + ChatGPT 5
// @license      GPLv3
// @match        https://codeforces.com/*
// @match        https://codeforc.es/*
// @run-at       document-idle
// @grant        GM_addStyle
// ==/UserScript==

(function () {
  "use strict";

  const normalize = () => {
    document.querySelectorAll('span.verdict-accepted').forEach((e) => {
      e.textContent = "CORRECT";
    });

    document.querySelectorAll('span.verdict-rejected').forEach((e) => {
      const t = e.textContent;
      if (/Wrong answer/i.test(t)) {
        e.textContent = "WRONG-ANSWER";
        e.className = "verdict-wrong";
      } else if (/Time limit/i.test(t)) {
        e.textContent = "TIMELIMIT";
        e.className = "verdict-tle";
      } else if (/Memory limit|Runtime error/i.test(t)) {
        e.textContent = "RUN-ERROR";
        e.className = "verdict-re";
      }
    });

    document.querySelectorAll('span[submissionverdict="COMPILATION_ERROR"]').forEach((e) => {
      e.textContent = "COMPILER-ERROR";
      e.className = "verdict-ce";
    });

    document.querySelectorAll('span.verdict-waiting').forEach((e) => {
      const t = e.textContent;
      if (/In queue/i.test(t)) {
        e.textContent = "PENDING";
        e.className = "verdict-pending";
      } else if (/Running/i.test(t)) {
        e.textContent = "JUDGING";
        e.className = "verdict-judging";
      } else {
        e.textContent = "PENDING";
        e.className = "verdict-pending";
      }
    });

    // 屏蔽时间和内存（提交列表）
    document.querySelectorAll('td.time-consumed-cell, td.memory-consumed-cell').forEach((e) => {
      e.textContent = "";
    });

    // 屏蔽提交详情页中的 "Time" / "Memory"
    document.querySelectorAll('div.property-time, div.property-memory').forEach((e) => {
      e.textContent = "";
    });
  };

  normalize();

  const submissions = document.querySelectorAll('tr[data-submission-id]');
  if (submissions.length > 0) {
    const observer = new MutationObserver(normalize);
    observer.observe(submissions[0].parentNode, { childList: true, subtree: true });
  }

  setInterval(() => {
    const element = document.getElementById("jGrowl");
    if (element?.parentNode) element.parentNode.removeChild(element);
  }, 200);

  // ===== DOMjudge 配色 =====
  GM_addStyle(`
    .verdict-accepted { color: #009900 !important; font-weight: bold; }  /* 绿色 */
    .verdict-wrong    { color: #ff0000 !important; font-weight: bold; }  /* 红色 */
    .verdict-tle      { color: #ff6600 !important; font-weight: bold; }  /* 橙色 */
    .verdict-re       { color: #9900cc !important; font-weight: bold; }  /* 紫色 */
    .verdict-ce       { color: #0066cc !important; font-weight: bold; }  /* 蓝色 */
    .verdict-pending  { color: #666666 !important; font-weight: bold; }  /* 灰色 */
    .verdict-judging  { color: #666666 !important; font-weight: bold; }  /* 灰色 */
  `);

})();
